#!/usr/bin/python
# -*- coding: utf-8 -*-

"""
Create a new note from the command-line
"""

import sys
import os
import getopt
import tempfile
import subprocess

def esc(arg):
  """
  Escapes a string's characters

  @param arg (string)
  @return string
  """
  buf = ''
  for i in arg:
    if (i == '\\'):
      buf += '\\\\'
    elif (i == '\n'):
      buf += '<br/>'
    elif (i == '"'):
      buf += '\\"'
    elif (i == ' '):
      buf += '&nbsp;'
    elif (i == '>'):
      buf += '&gt;'
    elif (i == '<'):
      buf += '&lt;'
    else:
      buf += i

  return buf
